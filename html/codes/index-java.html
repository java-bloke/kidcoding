/**
 * 判断术语词性是否为名词性
 *
 * @param term 术语
 * @return true=是名词性，false=不是名词性
 */
public static boolean isNameNature(Term term) {
    if (Objects.isNull(term)) {
        return false;
    }
    for(String n : N_TERM_NATURES) {
        if (term.nature.startsWith(n)) {
            return true;
        }
    }
    return false;
}
