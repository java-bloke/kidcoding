-- 按天投诉工单数
select 
date_format(t.date_created, '%Y-%m-%d') as '年-月-日', 
count(distinct t.order_no) as '进件工单数'
from mos.complaints_order t
where t.is_delete = 0
and t.date_created >= '2024-08-19 00:00:00'
-- and t.order_flag = 'UN'
group by DATE_FORMAT(t.date_created, '%Y-%m-%d')
order by DATE_FORMAT(t.date_created, '%Y-%m-%d')
;
