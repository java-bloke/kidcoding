// 1‌输出9x9乘法口诀表‌：这是一个基础示例，用于展示如何使用嵌套循环来打印出9x9的乘法口诀表。
#include <stdio.h>
int main() {
    int i, j, result;
    for (i = 1; i < 10; i++) {
        for (j = 1; j <= i; j++) {
            result = i * j;
            printf("%d*%d=%-4d", i, j, result);
        }
        printf("\n");
    }
    return 0;
}

// 2‌斐波那契数列‌：这是一个经典的算法问题，用于展示如何使用循环来计算斐波那契数列。
#include <stdio.h>
int main() {
    long f1, f2;
    int i;
    f1 = f2 = 1; // 初始化前两个斐波那契数
    for (i = 1; i <= 20; i++) {
        printf("%12ld %12ld", f1, f2);
        if (i % 2 == 0) printf("\n"); // 控制输出，每行四个数
        f1 = f1 + f2; // 前两个月加起来赋值给第三个月
        f2 = f1 + f2; // 前两个月加起来赋值给第三个月
    }
    return 0;
}
